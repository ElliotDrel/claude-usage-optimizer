    function statusColor(status: ScheduleRow['status']): string {
      switch (status) {
        case 'ok':      return 'var(--good)';
        case 'error':   return 'var(--danger)';
        case 'timeout': return 'var(--warning)';
        default:        return 'var(--text-tertiary)'; // pending

      if (isFiring) return 'Firing...';
      switch (status) {

        case 'error':   return 'Error';
        case 'timeout': return 'Timeout';
        default:        return 'Pending';
      }
    }

    function toLocalHHMM(utcIso: string): string {
      return new Date(utcIso).toLocaleTimeString(undefined, {
        hour: '2-digit',
        minute: '2-digit',
        hour12: false,
      });
    }

    4. Countdown formatter — takes delta in ms, returns formatted string:
    ```typescript
    function formatCountdown(deltaMs: number): string {
      if (deltaMs <= 0) return '0s';
      const totalSeconds = Math.ceil(deltaMs / 1000);
      const h = Math.floor(totalSeconds / 3600);
      const m = Math.floor((totalSeconds % 3600) / 60);
      const s = totalSeconds % 60;
      if (h > 0) return `${h}h ${m}m ${s}s`;
      if (m > 0) return `${m}m ${s}s`;
      return `${s}s`;
    }
    ```

    function FireSlotRow({

    6. NextFireCountdown sub-component — uses scheduledForUtc directly (no hour/minute parsing):
    ```typescript
    function NextFireCountdown({
      rows,
      now,
    }: {
      rows: ScheduleRow[];
      now: number;
    }) {
      // Find the next row: earliest pending row whose scheduledForUtc > now
      const nextRow = rows.find(
        (r) => r.status === 'pending' && new Date(r.scheduledForUtc).getTime() > now
      );

      if (!nextRow) {
        // All today's fires passed or no schedule — show nothing or tomorrow note
        return (
          <div className="mt-4 p-3 rounded-lg text-center" style={{ background: 'var(--bg-elevated)' }}>
            <p className="text-xs" style={{ color: 'var(--text-tertiary)' }}>
              All today&apos;s fires complete
            </p>
          </div>
        );
      }

      const deltaMs = new Date(nextRow.scheduledForUtc).getTime() - now;

      // "Firing..." if countdown <= 0 but row is still pending (waiting for next refresh)
      const isFiring = deltaMs <= 0;

      return (
        <div
          className="mt-4 p-4 rounded-lg text-center"
          style={{
            background: isFiring ? 'var(--warning-dim)' : 'var(--good-dim)',
            border: `1px solid ${isFiring ? 'var(--warning)' : 'var(--good)'}`,
          }}
        >
          <p className="text-[10px]" style={{ color: 'var(--text-tertiary)' }}>
            {isFiring ? 'Sending...' : 'Next fire in'}
          </p>
          <p
            className="text-2xl font-mono"
            style={{
              color: isFiring ? 'var(--warning)' : 'var(--good)',
              fontWeight: '600',
            }}
          >
            {isFiring ? 'Firing...' : formatCountdown(deltaMs)}
          </p>
        </div>
      );
    }
    ```

    ```typescript
    export function OptimalScheduleCard({ data }: { data: DashboardData | null }) {
      const [now, setNow] = useState(() => Date.now());
      const [activeTab, setActiveTab] = useState<'today' | 'tomorrow'>('today');

      useEffect(() => {
        const interval = setInterval(() => setNow(Date.now()), 1000);
        return () => clearInterval(interval); // cleanup prevents memory leak
      }, []);

      if (!data?.scheduleData) {
        return (
          <div style={{ padding: '20px', color: 'var(--text-tertiary)' }}>
            No schedule data available.
          </div>
        );
      }

      const { peakBlock, scheduleRows, tomorrowRows, scheduleGeneratedAt, isPaused } =
        data.scheduleData;

      const displayRows = activeTab === 'today' ? scheduleRows : tomorrowRows;

      return (
        <section
          className="rounded-xl p-5"
          style={{
            background: 'var(--bg-surface)',
            border: '1px solid var(--border-subtle)',
          }}
        >
          {/* Header: Peak Block + Tabs */}
          <div className="flex items-start justify-between gap-6 mb-6">
            <div>
              <h2
                className="text-lg"
                style={{
                  fontFamily: 'var(--font-display)',
                  color: 'var(--text-primary)',
                }}
              >
                Optimal Schedule
              </h2>
              {peakBlock ? (
                <p className="text-sm mt-2" style={{ color: 'var(--text-tertiary)' }}>
                  Peak:{' '}
                  <span style={{ color: 'var(--accent)', fontWeight: '600' }}>
                    {String(peakBlock.startHour).padStart(2, '0')}:00 –{' '}
                    {String(peakBlock.endHour).padStart(2, '0')}:00
                  </span>
                  , Midpoint{' '}
                  <span style={{ color: 'var(--accent)', fontWeight: '600' }}>
                    {String(peakBlock.midpoint).padStart(2, '0')}:00
                  </span>
                </p>
              ) : (
                <p className="text-sm mt-2" style={{ color: 'var(--text-tertiary)' }}>
                  No peak detected — using seed fallback
                </p>
              )}
            </div>

            {/* Tab buttons */}
            <div className="flex gap-2">
              {(['today', 'tomorrow'] as const).map((tab) => (
                <button
                  key={tab}
                  onClick={() => setActiveTab(tab)}
                  className="px-3 py-1.5 rounded text-xs font-medium uppercase tracking-wider"
                  style={{
                    background: activeTab === tab ? 'var(--accent)' : 'transparent',
                    color: activeTab === tab ? 'var(--bg-base)' : 'var(--text-secondary)',
                    border:
                      activeTab === tab
                        ? '1px solid var(--accent)'
                        : '1px solid var(--border-subtle)',
                    cursor: 'pointer',
                  }}
                >
                  {tab === 'today' ? 'Today' : 'Tomorrow'}
                </button>
              ))}
            </div>
          </div>

          {/* Paused banner */}
          {isPaused && (
            <div
              className="mb-4 px-3 py-2 rounded text-xs"
              style={{
                background: 'var(--warning-dim)',
                color: 'var(--warning)',
                border: '1px solid var(--warning)',
                fontFamily: 'var(--font-mono)',
              }}
            >
              Paused — Scheduled fires are skipped until resumed
            </div>
          )}

          {/* Fire slots */}
          {displayRows.length === 0 ? (
            <p className="text-sm" style={{ color: 'var(--text-tertiary)' }}>
              {scheduleGeneratedAt
                ? 'No fires scheduled today.'
                : 'Schedule not yet computed — runs at 03:00 UTC.'}
            </p>
          ) : (
            <div className="space-y-3">
              {displayRows.map((row) => (
                <FireSlotRow key={row.scheduledForUtc} row={row} now={now} />
              ))}
            </div>
          )}

          {/* Countdown (today tab only) */}
          {activeTab === 'today' && scheduleRows.length > 0 && (
            <NextFireCountdown rows={scheduleRows} now={now} />
          )}

          {/* Schedule generation timestamp */}
          <div className="mt-6 pt-4 border-t" style={{ borderColor: 'var(--border-subtle)' }}>
            <p className="text-[10px]" style={{ color: 'var(--text-tertiary)' }}>
              Schedule{' '}
              {scheduleGeneratedAt
                ? `updated ${new Date(scheduleGeneratedAt).toLocaleTimeString()}`
                : 'not yet computed'}
            </p>
          </div>
        </section>
      );
    }
    ```
