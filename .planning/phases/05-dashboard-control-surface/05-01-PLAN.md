    ```typescript
    const meta = getAppMeta(config);

    // Parse stored peak_block
    const peakBlockRaw = meta.get('peak_block') ?? '';
    const peakBlock = peakBlockRaw ? (JSON.parse(peakBlockRaw) as ScheduleData['peakBlock']) : null;

    // Parse stored schedule_fires — stored as { timestamp: string, isAnchor: boolean }[]
    const scheduledFiresRaw = meta.get('schedule_fires') ?? '[]';
    const scheduledFires: Array<{ timestamp: string; isAnchor: boolean }> =
      (() => { try { return JSON.parse(scheduledFiresRaw); } catch { return []; } })();

    // Build send_log lookup map: scheduled_for UTC timestamp -> SendLogRow
    const sendLogRows = querySendLog(config, { limit: 200, orderDesc: false });
    const sendLogByScheduledFor = new Map<string, typeof sendLogRows[0]>();
    for (const row of sendLogRows) {
      if (row.scheduled_for) {
        sendLogByScheduledFor.set(row.scheduled_for, row);
      }
    }

    // Compute "now" timestamp for isNext determination
    const nowMs = Date.now();

    // Derive schedule rows with status join
    const scheduleRows: ScheduleRow[] = scheduledFires.map((fire, idx) => {
      const logRow = sendLogByScheduledFor.get(fire.timestamp);
      const status: ScheduleRow['status'] = logRow
        ? (logRow.status as 'ok' | 'error' | 'timeout')
        : 'pending';
      // displayTime: convert UTC ISO to local HH:MM using browser-side toLocaleTimeString
      // On server, return the UTC time string raw — client will reformat via toLocaleTimeString
      const displayTime = new Date(fire.timestamp).toISOString().slice(11, 16); // "HH:MM" UTC fallback
      return {
        scheduledForUtc: fire.timestamp,
        displayTime,
        isAnchor: fire.isAnchor,
        status,
        isNext: false, // set below
      };
    });

    // Set isNext: earliest pending row with scheduledForUtc > now
    const pendingFuture = scheduleRows.filter(
      (r) => r.status === 'pending' && new Date(r.scheduledForUtc).getTime() > nowMs
    );
    if (pendingFuture.length > 0) {
      const nextTs = pendingFuture[0].scheduledForUtc;
      for (const row of scheduleRows) {
        if (row.scheduledForUtc === nextTs) row.isNext = true;
      }
    }

    // Tomorrow's fires: shift each scheduled_for timestamp by +24h, status always 'pending'
    const tomorrowRows: ScheduleRow[] = scheduledFires.map((fire) => {
      const tomorrowTs = new Date(new Date(fire.timestamp).getTime() + 24 * 60 * 60 * 1000).toISOString();
      const displayTime = new Date(tomorrowTs).toISOString().slice(11, 16);
      return {
        scheduledForUtc: tomorrowTs,
        displayTime,
        isAnchor: fire.isAnchor,
        status: 'pending' as const,
        isNext: false,
      };
    });

    // Build send history (last 20 rows, most recent first)
    const historyRows = querySendLog(config, { limit: 20, orderDesc: true });
    const sendHistory: SendLogEntry[] = historyRows.map((row) => ({
      id: row.id,
      firedAt: row.fired_at,
      scheduledFor: row.scheduled_for,
      isAnchor: row.is_anchor === 1,
      status: row.status as 'ok' | 'error' | 'timeout',
      durationMs: row.duration_ms,
      question: row.question,
      responseExcerpt: row.response_excerpt,
    }));

    // Collect schedule config values
    const scheduleData: ScheduleData = {
      peakBlock,
      scheduleRows,
      tomorrowRows,
      scheduleGeneratedAt: meta.get('schedule_generated_at') ?? null,
      isPaused: meta.get('paused') === 'true',
      overrideStartTime: meta.get('schedule_override_start_time') ?? '',
      peakWindowHours: meta.get('peak_window_hours') ?? '4',
      anchorOffsetMinutes: meta.get('anchor_offset_minutes') ?? '5',
      defaultSeedTime: meta.get('default_seed_time') ?? '05:05',
      userTimezone: meta.get('user_timezone') ?? 'America/Los_Angeles',
    };
    ```

    ```typescript
    import { NextResponse, type NextRequest } from "next/server";
    import { getConfig } from "@/lib/config";
    import { setAppMeta } from "@/lib/db";
    import { recomputeSchedule } from "@/lib/scheduler";

    export const dynamic = "force-dynamic";

    // Keys the client may write
    const ALLOWED_KEYS = new Set([
      "schedule_override_start_time",
      "peak_window_hours",
      "anchor_offset_minutes",
      "default_seed_time",
      "user_timezone",
      "paused",
    ]);

    // Keys that change the schedule output — trigger recomputeSchedule after write.
    // "paused" is intentionally excluded: recomputeSchedule() resets schedule_fires_done
    // to [], which would re-enable already-fired sends if called mid-day.
    const SCHEDULE_AFFECTING_KEYS = new Set([
      "schedule_override_start_time",
      "peak_window_hours",
      "anchor_offset_minutes",
      "default_seed_time",
      "user_timezone",
    ]);

    // Regex for HH:MM time format
    const HHMM_RE = /^\d{2}:\d{2}$/;

    /**
     * validateValue — returns an error string if value is invalid for this key,
     * or null if valid.
     */
    function validateValue(key: string, value: string): string | null {
      switch (key) {
        case "peak_window_hours": {
          const n = parseInt(value, 10);
          if (isNaN(n) || n < 3 || n > 6 || String(n) !== value.trim()) {
            return "peak_window_hours must be an integer between 3 and 6";
          }
          return null;
        }
        case "anchor_offset_minutes": {
          const n = parseInt(value, 10);
          if (isNaN(n) || n < 0 || n > 15 || String(n) !== value.trim()) {
            return "anchor_offset_minutes must be an integer between 0 and 15";
          }
          return null;
        }
        case "schedule_override_start_time": {
          // Allow empty string (clears override) or HH:MM format
          if (value !== "" && !HHMM_RE.test(value)) {
            return "schedule_override_start_time must be HH:MM format or empty string";
          }
          return null;
        }
        case "default_seed_time": {
          if (!HHMM_RE.test(value)) {
            return "default_seed_time must be HH:MM format";
          }
          return null;
        }
        case "user_timezone": {
          // Must be a valid IANA timezone name — same validation scheduler.ts uses
          try {
            Intl.DateTimeFormat("en-US", { timeZone: value });
            return null;
          } catch {
            return `user_timezone must be a valid IANA timezone name (e.g. America/New_York), got: ${value}`;
          }
        }
        case "paused": {
          if (value !== "true" && value !== "false") {
            return "paused must be exactly 'true' or 'false'";
          }
          return null;
        }
        default:
          return null; // Unknown keys are blocked by ALLOWED_KEYS check above
      }
    }

    export async function PATCH(req: NextRequest) {
      try {
        const body = (await req.json()) as { key?: string; value?: string };

        if (!body.key || body.value === undefined) {
          return NextResponse.json({ error: "Missing key or value" }, { status: 400 });
        }

        if (!ALLOWED_KEYS.has(body.key)) {
          return NextResponse.json(
            { error: `Key '${body.key}' is not allowed. Allowed keys: ${[...ALLOWED_KEYS].join(", ")}` },
            { status: 400 }
          );
        }

        const validationError = validateValue(body.key, body.value);
        if (validationError) {
          return NextResponse.json({ error: validationError }, { status: 400 });
        }

        const config = getConfig();

        // Write the validated value
        setAppMeta(config, body.key, body.value);

        // Only recompute for schedule-affecting keys — NOT for 'paused'.
        // recomputeSchedule() resets schedule_fires_done to [], which would
        // re-enable already-fired sends if called mid-day on a pause toggle.
        if (SCHEDULE_AFFECTING_KEYS.has(body.key)) {
          recomputeSchedule(config);
        }

        return NextResponse.json({
          success: true,
          key: body.key,
          value: body.value,
        });
      } catch (err) {
        console.error("[app-meta PATCH]", err);
        return NextResponse.json(
          { error: err instanceof Error ? err.message : "Unknown error" },
          { status: 500 }
        );
      }
    }
    ```
